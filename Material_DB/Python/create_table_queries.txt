   eps_r_Fkt_Typ         INT,
   eps_r_fmax         REAL,
   eps_r_a         REAL,
   eps_r_b         REAL,
   eps_r_c         REAL,
   kappa_fmin        REAL,
   kappa_Fkt_Typ         INT,
   kappa_fmax         REAL,
   kappa_a         REAL,
   kappa_b         REAL,
   kappa_c         REAL,
   my_r_fmin        REAL,
   my_r_Fkt_Typ         INT,
   my_r_fmax         REAL,
   my_r_a         REAL,
   my_r_b         REAL,
   my_r_c         REAL,
   f_min         REAL,
   f_max         REAL
);
